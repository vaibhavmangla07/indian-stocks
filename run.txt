### Docker commands

# Build the Docker image
docker build -t stocksy:latest .

# Run the container on port 8501
# On macOS, host.docker.internal lets the container reach Ollama running on your host machine.
docker run --rm -p 8501:8501 \
  --env-file .env \
  -e OLLAMA_BASE_URL=http://host.docker.internal:11434 \
  -e OLLAMA_MODEL=gemma3:270m \
  stocksy:latest
